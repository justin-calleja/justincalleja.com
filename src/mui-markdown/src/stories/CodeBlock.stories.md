### Javascript

```jsx
const user = 'Mopeim';

const sayHello = (name) => {
  console.log(`Hello ${name}`);
};

sayHello(user); // Result: Hello Mopeim
```

### Typescript

```tsx
const user = 'Mopeim';

const sayHello = (name: string) => {
  console.log(`Hello ${name}`);
};

sayHello(user); // Result: Hello Mopeim
```
